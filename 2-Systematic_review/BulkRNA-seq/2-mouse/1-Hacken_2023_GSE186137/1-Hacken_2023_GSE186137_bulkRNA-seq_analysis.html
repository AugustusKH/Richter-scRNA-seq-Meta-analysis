Hacken_2023_GSE186137_bulkRNA-seq_analysis

Author

Pakorn Sagulkoo

In this code, we will analyse mouse bulk RNA-seq from Hacken et al. (2023). This study employed a mouse experimental model in which a CLL- and RT-like condition was induced using del(13q) Cd19-Cas9-GFP mice engrafted with LSK cells transduced with sgRNA-expressing lentivirus, followed by multiple loss-of-function alterations. After we performed Salmon converting FASTQ files to pseudocount data in each sample, we will convert them to a count matrix before we use the matrix for DESeq2 for further gene expression analysis.

Create a DESeq2 object

# Create metadata for DESeq2 object
colData <- sample_metadata[, c("SRA_RUN", "condition")] %>%
  rename(samples = "SRA_RUN") 
colData$group <- ifelse(colData$condition == "RT", "RT", "Others")
rownames(colData) <- colData$samples
colData$condition <- factor(colData$condition)
DT::datatable(colData, options = list(scrollX = TRUE))
# Create the object
dds <- DESeqDataSetFromTximport(txi, colData, ~ condition)
# Keep genes with at least 10 counts across all samples
keep <- rowSums(counts(dds)) >= 10
dds <- dds[keep,]
  
# Run DESeq2
dds$condition <- relevel(dds$condition, ref = "CTRL")
dds <- DESeq(dds)

Plot the PCA based on the expression data

# Variance stabilised transformation
vsdata <- DESeq2::vst(dds, blind = FALSE)

# Extract matrix
vs_mat <- assay(vsdata)

# Adjust for batch (assume batch info is in colData(dds)$batch)
vs_mat_corrected <- removeBatchEffect(vs_mat, batch = colData(dds)$batch)

# PCA on corrected data
pca <- prcomp(t(vs_mat_corrected))

# Make a dataframe for ggplot
pca_df <- data.frame(pca$x,
                     condition = colData(dds)$condition,
                     sample = rownames(colData(dds)))

# Plot
ggplot(pca_df, aes(PC1, PC2, color = condition)) +
  geom_point(size = 3) +
  theme_bw()

Plot a heatmap of gene expression

# Import the relevant package
library(pheatmap)
library(RColorBrewer)
# Data transformation
vsd <- DESeq2::vst(dds, blind=FALSE)       # Variance-stabilizing transformation
#rld <- rlog(dds_filtered, blind=FALSE)      # Regularized log transformation

# Extract transformed expression matrix
expr_mat <- assay(vsd)

# Calculate z-scores
zscore_mat <- t(scale(t(expr_mat)))
# Define samples in each condition
CTRL_sample <- colData$samples[colData$condition == "CTRL"]
CLL_sample <- colData$samples[colData$condition == "CLL"]
CLL_RT_sample <- colData$samples[colData$condition == "CLL/RT"]
RT_sample <- colData$samples[colData$condition == "RT"]
# Extract phenotypes info from column names
conditions <-  ifelse(colnames(zscore_mat) %in% CTRL_sample, "CTRL", 
                      ifelse(colnames(zscore_mat) %in% CLL_sample, "CLL", 
                             ifelse(colnames(zscore_mat) %in% CLL_RT_sample, "CLL/RT", "RT")))

# Create annotation dataframe
annotation_col <- data.frame(
  Condition = factor(conditions, levels = c("CTRL", "CLL", "CLL/RT", "RT"))
)
annotation_col <- annotation_col[, "Condition", drop = FALSE]  # keep as data frame
rownames(annotation_col) <- colnames(zscore_mat)

# Order by Condition and then Case 
order_cols <- order(annotation_col$Condition)
zscore_mat_ordered <- zscore_mat[, order_cols]
annotation_col_ordered <- annotation_col[order_cols, , drop = FALSE]
# Filter for significant DE genes
sig_genes <- human_res_RT_vs_others[!is.na(human_res_RT_vs_others$padj) & human_res_RT_vs_others$padj < 0.05 & abs(human_res_RT_vs_others$log2FoldChange) > 2, ]

# Order genes by log2FoldChange (descending: up -> down)
gene_order <- sig_genes$Mouse_Ensembl_ID[order(sig_genes$log2FoldChange, decreasing = FALSE)]

# Subset zscore matrix
zscore_mat_sig <- zscore_mat_ordered[gene_order, ]

# Plot heatmap
pheatmap(
  zscore_mat_sig,
  cluster_rows = FALSE,
  cluster_cols = FALSE,
  show_rownames = FALSE,
  color = colorRampPalette(rev(brewer.pal(11, "RdYlBu")))(100),
  annotation_col = annotation_col_ordered,
  fontsize = 8,
  fontsize_row = 7,
  fontsize_col = 7,
  gaps_col = which(diff(as.numeric(factor(annotation_col_ordered$Condition))) != 0)
)
